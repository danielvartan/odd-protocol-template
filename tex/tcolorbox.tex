\newcommand{\arrowbullet}[1][]{
  \begin{tikzpicture}[#1]
    \draw[-{Triangle[width=6.75pt,length=3pt]},
    line width=3pt](0,0) -- (0.3, 0);
  \end{tikzpicture}
}

\newcommand{\infoicon}[1][]{
\begin{tikzpicture}[#1]
  \draw[line width=1.5pt] (0,0) circle(0.2cm);
  \node at (0,0) {\fontsize{8pt}{8pt}\selectfont\textbf{i}};
\end{tikzpicture}
}
